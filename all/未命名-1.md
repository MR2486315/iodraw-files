from graphviz import Digraph

dot = Digraph('health_arch', format='png', encoding='utf-8')
dot.attr(rankdir='TB', size='14,18', dpi='300', fontname='Microsoft YaHei')
dot.attr('node', shape='rect', style='filled', fontname='Microsoft YaHei', fontsize='11')

# 表现层
with dot.subgraph(name='cluster_0') as c:
    c.attr(label='1. 表现层 Presentation Layer', color='#2E86AB', fontsize='14', penwidth='2')
    c.attr(bgcolor='#EFF8FF')
    c.node('web', 'PC端Web界面\nReact18 + TypeScript + Vite', fillcolor='#D6EAF8')
    c.node('dash', '数据看板\nRecharts趋势图', fillcolor='#D6EAF8')
    c.node('chat', '咨询对话框\n即时通讯组件', fillcolor='#D6EAF8')
    c.node('admin', '管理后台\n用户管理&仪表盘', fillcolor='#D6EAF8')
    c.edge('web', 'dash')
    c.edge('web', 'chat')
    c.edge('web', 'admin')

# 接口层
with dot.subgraph(name='cluster_1') as c:
    c.attr(label='2. 接口服务层 API Layer', color='#A23B72', fontsize='14', penwidth='2')
    c.attr(bgcolor='#FFF5FB')
    c.node('rest', 'RESTful HTTP/HTTPS', fillcolor='#FADBD8')
    c.node('auth', '安全机制\nJWT认证 + API Key校验', fillcolor='#FADBD8')
    c.node('ctrl', '核心控制器\n居民/健康/咨询Controller', fillcolor='#FADBD8')

# 业务层
with dot.subgraph(name='cluster_2') as c:
    c.attr(label='3. 业务逻辑层 Business Logic Layer', color='#F18F01', fontsize='14', penwidth='2')
    c.attr(bgcolor='#FFF9E9')
    c.node('hp', '健康数据处理\n指标提取&历史趋势分析', fillcolor='#FDEBCA')
    c.node('ai', 'AI决策中心\nLLM+RAG检索增强生成', fillcolor='#FDEBCA')
    c.node('risk', '风险评估模型\n自动分级&健康预警', fillcolor='#FDEBCA')

# 数据层
with dot.subgraph(name='cluster_3') as c:
    c.attr(label='4. 数据访问层 Data Access Layer', color='#C73E1D', fontsize='14', penwidth='2')
    c.attr(bgcolor='#FFF0EE')
    c.node('orm', 'ORM：SQLAlchemy', fillcolor='#FADBD8')
    c.node('model', '核心实体模型\n居民/健康记录/对话模型', fillcolor='#FADBD8')

# 基础设施层
with dot.subgraph(name='cluster_4') as c:
    c.attr(label='5. 基础设施层 Infrastructure Layer', color='#005A8D', fontsize='14', penwidth='2')
    c.attr(bgcolor='#EDF7FF')
    c.node('db', 'MySQL8.0 + Redis\n持久化&缓存会话', fillcolor='#D6EAF8')
    c.node('ext', '外部集成\nDify API + DeepSeek大模型', fillcolor='#D6EAF8')
    c.node('deploy', '部署环境\nDocker & Docker-Compose', fillcolor='#D6EAF8')

# 调用关系
dot.edge('dash', 'rest', dir='forward')
dot.edge('chat', 'rest', dir='forward')
dot.edge('admin', 'rest', dir='forward')

dot.edge('rest', 'ctrl')
dot.edge('auth', 'ctrl')
dot.edge('ctrl', 'hp')
dot.edge('ctrl', 'ai')
dot.edge('ctrl', 'risk')

dot.edge('hp', 'orm')
dot.edge('ai', 'orm')
dot.edge('risk', 'orm')
dot.edge('ai', 'ext')

dot.edge('orm', 'model')
dot.edge('model', 'db')

dot.render('health_management_arch', view=False)
print("架构图已生成：health_management_arch.png")
